import floatPlugin from 'tailwindcss/lib/plugins/float'

export const classes = {
  plugin: floatPlugin,
  transformSelector: (selector) => {
    if (selector === '.clearfix:after') {
      return '.clearfix'
    }
    return selector
  },
  transformProperties: ({ selector, properties }) => {
    if (selector === '.clearfix:after') {
      return {
        '&::after': properties,
      }
    }
    return properties
  },
}
