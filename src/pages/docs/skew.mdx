import utilities from 'utilities?plugin=skew'
import { numbersFirst } from '@/utils/sortClasses'
import { ArbitraryValues } from '@/components/ArbitraryValues'

import { HoverFocusAndOtherStates } from '@/components/HoverFocusAndOtherStates'
import { RemovingTransforms } from '@/components/RemovingTransforms'

export const classes = {
  utilities,
  sort: (classes) => numbersFirst(classes),
  transformProperties: ({ properties }) => {
    let x = properties['--tw-skew-x']
    let y = properties['--tw-skew-y']
    return {
      transform: `skew${x ? 'X' : 'Y'}(${x || y})`
    }
  },
}

### Using negative values

To use a negative skew value, prefix the class name with a dash to convert it to a negative value.

```html
<img class="**-skew-y-6** ...">
```

## Using custom values

### Customizing your theme

By default, Tailwind includes a handful of general purpose `skew` utilities. You can customize these values by editing `theme.skew` or `theme.extend.skew` in your `tailwind.config.js` file.

```diff-js tailwind.config.js
  module.exports = {
    theme: {
      extend: {
+       skew: {
+         '17': '17deg',
+       }
      }
    }
  }
```

Learn more about customizing the default theme in the [theme customization](/docs/theme#customizing-the-default-theme) documentation.
